\begin{align}
    \mathrm{Accuracy}\, (\mathrm{Acc})      &= \frac{TP + TN}{TP + TN + FP + FN} \label{eq:AE_accuracy} \\[1ex]
    \mathrm{Sensitivity}\, (\mathrm{Sn})    &= \frac{TP}{TP + FN} \label{eq:AE_sensitivity} \\[1ex]
    \mathrm{Specificity}\, (\mathrm{Sp})    &= \frac{TN}{TN + FP} \label{eq:AE_specificity} \\[1ex]
    \mathrm{F1}                             &= \frac{2 \cdot TP}{2 \cdot TP + FP + FN} \label{eq:AE_f1} \\[1ex]
    \mathrm{ROC\text{-}AUC}                 &= P\big(a(x_i) < a(x_j) \mid y_i = 0,\, y_j = 1 \big) \label{eq:AE_roc_auc} \\[1ex]
    \mathrm{ROC\text{-}AUC}\ 95\%\, \mathrm{CI} &= \hat{A} \pm z \cdot \sqrt{\frac{\operatorname{Var}(X) + \operatorname{Var}(Y)}{n_0 \times n_1}} \label{eq:AE_roc_auc_ci}
\end{align}
