\node[data]     (raw)    {Raw Data\\(CSV, JSON)};
\node[proc, below=of raw] (clean)  {Clean \&\\Validate};
\node[proc, below=of clean] (feat) {Feature\\Extraction};
\node[decision, below=of feat] (qc) {QC\\Pass?};
\node[box, below=of qc] (model) {Train\\Model};
\node[proc, below=of model] (eval) {Evaluate};
\node[box, below=of eval] (report) {Report\\Results};

\node[proc, left=18mm of qc] (fix) {Adjust\\Parameters};

\draw[arrow] (model) -- (eval);
\draw[arrow] (eval) -- (report);

\draw[arrow] (qc.west) -- node[above, font=\scriptsize]{no} (fix.east);
\draw[arrow] (fix.north) |- (clean.west);

\node[inner sep=0pt, fit=(clean)(feat)(qc)] (prepblock) {};

% Fit group for modeling stage

      label={[font=\scriptsize]right:Modeling stage}] {};
